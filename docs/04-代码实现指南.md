### 3.2 LLM 初始化 (app/core/llm.py)

```python
import os
from langchain_openai import ChatOpenAI
from langchain_community.embeddings import HuggingFaceEmbeddings

def get_llm():
    """获取 LLM 实例"""
    return ChatOpenAI(
        model=os.getenv("LLM_MODEL", "qwen-max"),
        temperature=float(os.getenv("LLM_TEMPERATURE", "0.7")),
        max_tokens=int(os.getenv("LLM_MAX_TOKENS", "2000")),
        api_key=os.getenv("DASHSCOPE_API_KEY"),
        base_url="https://dashscope.aliyuncs.com/compatible-mode/v1",
    )

def get_embedding():
    """获取 Embedding 实例"""
    return HuggingFaceEmbeddings(
        model_name="BAAI/bge-small-zh-v1.5",
        model_kwargs={'device': 'cpu'},
        encode_kwargs={'normalize_embeddings': True}
    )
```

### 3.3 文档加载 (app/rag/document_loader.py)

```python
from langchain_community.document_loaders import (
    PyPDFLoader,
    Docx2txtLoader,
    TextLoader,
    UnstructuredExcelLoader,
)
from pathlib import Path

def load_document(file_path: str):
    """根据文件类型选择加载器"""
    path = Path(file_path)
    suffix = path.suffix.lower()
    
    loaders = {
        '.pdf': PyPDFLoader,
        '.docx': Docx2txtLoader,
        '.doc': Docx2txtLoader,
        '.txt': TextLoader,
        '.xlsx': UnstructuredExcelLoader,
        '.xls': UnstructuredExcelLoader,
    }
    
    loader_class = loaders.get(suffix)
    if not loader_class:
        raise ValueError(f"不支持的文件类型: {suffix}")
    
    # 统一使用 UTF-8 编码
    if suffix == '.txt':
        loader = loader_class(file_path, encoding='utf-8')
    else:
        loader = loader_class(file_path)
    
    return loader.load()
```

### 3.4 文本分块 (app/rag/text_splitter.py)

```python
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.schema import Document

def get_text_splitter():
    """获取文本分块器"""
    return RecursiveCharacterTextSplitter(
        chunk_size=512,
        chunk_overlap=50,
        length_function=len,
        separators=[
            "\n\n",   # 段落分隔
            "\n",     # 行分隔
            "。",     # 句子分隔（中文）
            "！",
            "？",
            ".",
            "!",
            "?",
            " ",      # 词分隔
            "",
        ],
    )

def split_documents(documents: list[Document]):
    """分块文档"""
    splitter = get_text_splitter()
    return splitter.split_documents(documents)
```

### 3.5 向量存储 (app/rag/vector_store.py)

```python
from langchain_community.vectorstores import Chroma
from langchain.schema import Document
from app.core.llm import get_embedding

def create_vector_store(documents: list[Document], persist_directory: str = "./storage/chroma"):
    """创建向量存储"""
    embedding = get_embedding()
    
    # 提取文档内容用于去重
    texts = [doc.page_content for doc in documents]
    metadatas = [doc.metadata for doc in documents]
    
    # 创建 Chroma 向量库
    vector_store = Chroma.from_texts(
        texts=texts,
        embedding=embedding,
        metadatas=metadatas,
        persist_directory=persist_directory,
    )
    
    return vector_store

def load_vector_store(persist_directory: str = "./storage/chroma"):
    """加载已有向量存储"""
    embedding = get_embedding()
    return Chroma(
        persist_directory=persist_directory,
        embedding_function=embedding,
    )
```

### 3.6 检索器 (app/rag/retriever.py)

```python
from langchain.schema import Document

def create_retriever(vector_store, top_k: int = 5, score_threshold: float = 0.7):
    """创建检索器"""
    return vector_store.as_retriever(
        search_type="similarity_score_threshold",
        search_kwargs={
            "k": top_k,
            "score_threshold": score_threshold,
        }
    )

def retrieve_documents(retriever, question: str) -> list[Document]:
    """检索相关文档"""
    return retriever.get_relevant_documents(question)
```

### 3.7 Prompt 模板 (app/rag/prompt.py)

```python
from langchain.prompts import ChatPromptTemplate

# 系统提示词
SYSTEM_PROMPT = """你是企业 HR 智能助手，负责解答员工关于公司政策的问题。

## 约束
1. 只基于提供的参考文档回答，不要编造信息
2. 回答要准确引用来源，无法回答时明确说明"未找到相关信息，建议咨询 HR 部门"
3. 回答格式：结论 + 依据来源
4. 保持专业、友好的语气

## 回答格式
[回答内容]

📄 参考来源：[文档名] (页码)
"""

# 用户提示词模板
USER_PROMPT_TEMPLATE = """基于以下参考文档回答用户问题：

=== 参考文档 ===
{context}
===

用户问题：{question}

请根据参考文档回答，如果文档中没有相关信息，请明确说明。"""

def get_qa_prompt():
    """获取问答提示词模板"""
    return ChatPromptTemplate.from_messages([
        ("system", SYSTEM_PROMPT),
        ("human", USER_PROMPT_TEMPLATE),
    ])

def format_context(documents: list[Document]) -> str:
    """格式化文档为上下文"""
    context_parts = []
    for i, doc in enumerate(documents, 1):
        source = doc.metadata.get("source", "未知来源")
        page = doc.metadata.get("page", "")
        content = doc.page_content
        
        context_parts.append(f"【文档 {i}】来源：{source} (页码：{page})\n{content}")
    
    return "\n\n".join(context_parts)
```

### 3.8 对话 API (app/api/chat.py)

```python
from fastapi import APIRouter, HTTPException
from pydantic import BaseModel
from app.core.llm import get_llm
from app.rag.vector_store import load_vector_store
from app.rag.retriever import create_retriever
from app.rag.prompt import get_qa_prompt, format_context

router = APIRouter(prefix="/api/v1", tags=["chat"])

class ChatRequest(BaseModel):
    question: str
    session_id: str = None

class ChatResponse(BaseModel):
    answer: str
    sources: list[dict]

@router.post("/chat", response_model=ChatResponse)
async def chat(request: ChatRequest):
    """HR 问答接口"""
    try:
        # 1. 加载向量库
        vector_store = load_vector_store()
        
        # 2. 创建检索器
        retriever = create_retriever(vector_store, top_k=5, score_threshold=0.7)
        
        # 3. 检索相关文档
        docs = retriever.get_relevant_documents(request.question)
        
        if not docs:
            return ChatResponse(
                answer="抱歉，我未找到相关信息。建议您联系 HR 部门获取更准确的解答。",
                sources=[]
            )
        
        # 4. 格式化上下文
        context = format_context(docs)
        
        # 5. 构建提示词
        prompt = get_qa_prompt()
        messages = prompt.format_messages(
            context=context,
            question=request.question
        )
        
        # 6. 调用 LLM 生成回答
        llm = get_llm()
        response = llm.invoke(messages)
        
        # 7. 提取来源信息
        sources = []
        for doc in docs:
            sources.append({
                "doc": doc.metadata.get("source", "未知"),
                "page": doc.metadata.get("page", ""),
                "section": doc.metadata.get("section", ""),
            })
        
        return ChatResponse(
            answer=response.content,
            sources=sources
        )
        
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
```

### 5.2 初始化知识库

```python
# 在项目根目录执行
python -c "

from app.rag.vector_store import create_vector_store
```
